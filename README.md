def shuffle_deck(deck: list) -> list:
shuffle(deck)
return deck

Функция для раздачи карт
def draw_card(deck, num_cards=1):
drawn_cards = []
for _ in range(num_cards):
if deck: # Проверяем, что колода не пуста
card = deck.pop()
drawn_cards.append(card)
else:
print("Колода пуста!")
break
return drawn_cards
